class UrbanAI { constructor() { this.apiBase = '/api'; this.isInitialized = false; this.documentsProcessed = false; this.initializeElements(); this.attachEventListeners(); this.autoInitialize(); // AUTO-START! } initializeElements() { this.elements = { initBtn: document.getElementById('initBtn'), processBtn: document.getElementById('processBtn'), summaryBtn: document.getElementById('summaryBtn'), keyPointsBtn: document.getElementById('keyPointsBtn'), chatMessages: document.getElementById('chatMessages'), chatInput: document.getElementById('chatInput'), sendBtn: document.getElementById('sendBtn'), status: document.getElementById('status'), typing: document.getElementById('typing') }; } attachEventListeners() { this.elements.initBtn.addEventListener('click', () => this.manualInit()); this.elements.processBtn.addEventListener('click', () => this.processDocuments()); this.elements.summaryBtn.addEventListener('click', () => this.getDocumentSummary()); this.elements.keyPointsBtn.addEventListener('click', () => this.getKeyPoints()); this.elements.sendBtn.addEventListener('click', () => this.sendMessage()); this.elements.chatInput.addEventListener('keydown', (e) => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); this.sendMessage(); } }); // Auto-resize textarea this.elements.chatInput.addEventListener('input', () => { this.elements.chatInput.style.height = 'auto'; this.elements.chatInput.style.height = this.elements.chatInput.scrollHeight + 'px'; }); } updateStatus(text, type = 'loading') { const statusEl = this.elements.status; statusEl.className = `status ${type}`; statusEl.innerHTML = `${text}`; } addMessage(content, isUser = false) { const welcomeEl = document.querySelector('.welcome'); if (welcomeEl) welcomeEl.remove(); const messageDiv = document.createElement('div'); messageDiv.className = `message ${isUser ? 'user' : 'bot'}`; const avatar = document.createElement('div'); avatar.className = `avatar ${isUser ? 'user' : 'bot'}`; avatar.textContent = isUser ? 'U' : 'AI'; const contentDiv = document.createElement('div'); contentDiv.className = 'message-content'; contentDiv.innerHTML = content.replace(/\n/g, '
'); messageDiv.appendChild(avatar); messageDiv.appendChild(contentDiv); this.elements.chatMessages.appendChild(messageDiv); this.elements.chatMessages.scrollTop = this.elements.chatMessages.scrollHeight; } showTyping(show = true) { this.elements.typing.classList.toggle('show', show); if (show) { this.elements.chatMessages.scrollTop = this.elements.chatMessages.scrollHeight; } } async makeRequest(endpoint, options = {}) { const response = await fetch(`${this.apiBase}${endpoint}`, { headers: { 'Content-Type': 'application/json' }, ...options }); if (!response.ok) { throw new Error(`HTTP ${response.status}`); } return await response.json(); } async checkSystemStatus() { try { await this.makeRequest('/health'); this.updateStatus('✅ Connesso', 'ready'); return true; } catch (error) { this.updateStatus('❌ Errore connessione', 'error'); return false; } } // AUTO INITIALIZATION async autoInitialize() { this.updateStatus('🚀 Caricamento automatico...', 'loading'); try { // Step 1: Check connection const connected = await this.checkSystemStatus(); if (!connected) throw new Error('Connessione fallita'); // Step 2: Auto-initialize await this.makeRequest('/initialize', { method: 'POST' }); this.isInitialized = true; this.updateStatus('✅ Sistema pronto', 'ready'); // Enable buttons this.elements.processBtn.disabled = false; this.elements.summaryBtn.disabled = false; this.elements.keyPointsBtn.disabled = false; this.elements.chatInput.disabled = false; this.elements.sendBtn.disabled = false; // Hide manual init button this.elements.initBtn.style.display = 'none'; this.addMessage('✅ Sistema pronto! Puoi fare domande direttamente o usare i pulsanti per riassunti e punti chiave.'); } catch (error) { this.updateStatus('❌ Errore avvio', 'error'); this.addMessage('❌ Errore nell\'inizializzazione automatica. Usa il pulsante "Inizializza Sistema".'); } } // Manual fallback async manualInit() { try { this.updateStatus('⚡ Inizializzando...', 'loading'); await this.makeRequest('/initialize', { method: 'POST' }); this.isInitialized = true; this.updateStatus('✅ Sistema pronto', 'ready'); this.elements.processBtn.disabled = false; this.elements.summaryBtn.disabled = false; this.elements.keyPointsBtn.disabled = false; this.elements.chatInput.disabled = false; this.elements.sendBtn.disabled = false; this.elements.initBtn.style.display = 'none'; this.addMessage('Sistema inizializzato manualmente!'); } catch (error) { this.updateStatus('❌ Errore inizializzazione', 'error'); this.addMessage('Errore: ' + error.message); } } async ensureReady() { if (!this.isInitialized) { await this.autoInitialize(); } } async sendMessage() { await this.ensureReady(); const question = this.elements.chatInput.value.trim(); if (!question) return; this.elements.chatInput.value = ''; this.elements.chatInput.style.height = 'auto'; this.addMessage(question, true); this.showTyping(true); try { const result = await this.makeRequest('/query', { method: 'POST', body: JSON.stringify({ question }) }); this.showTyping(false); this.addMessage(result.answer); } catch (error) { this.showTyping(false); this.addMessage('Errore: ' + error.message); } } async processDocuments() { await this.ensureReady(); this.updateStatus('📁 Elaborando documenti...', 'loading'); this.addMessage('Elaborazione documenti in corso...'); try { const result = await this.makeRequest('/process-documents', { method: 'POST' }); this.documentsProcessed = true; this.updateStatus('✅ Documenti elaborati', 'ready'); this.addMessage(`Documenti elaborati! Processati ${result.chunksProcessed} frammenti.`); } catch (error) { this.updateStatus('❌ Errore elaborazione', 'error'); this.addMessage('Errore: ' + error.message); } } async getDocumentSummary() { await this.ensureReady(); this.showTyping(true); try { const result = await this.makeRequest('/summarize'); this.showTyping(false); this.addMessage(`📋 **Riassunto Documenti:**\n\n${result.summary}`); } catch (error) { this.showTyping(false); this.addMessage('Errore nel riassunto: ' + error.message); } } async getKeyPoints() { await this.ensureReady(); const topic = prompt('Su quale argomento vuoi i punti chiave?', 'pianificazione urbanistica'); if (!topic) return; this.showTyping(true); try { const result = await this.makeRequest('/extract-key-points', { method: 'POST', body: JSON.stringify({ topic }) }); this.showTyping(false); this.addMessage(`🎯 **Punti Chiave - ${topic}:**\n\n${result.keyPoints}`); } catch (error) { this.showTyping(false); this.addMessage('Errore: ' + error.message); } } } // Initialize app document.addEventListener('DOMContentLoaded', () => { new UrbanAI(); });
